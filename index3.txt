<script>
    // Dữ liệu mẫu câu
    const targetText = "你好很高兴认识你"; // Chuỗi chuẩn để so sánh (bỏ dấu câu)
    
    // 1. Phát giọng mẫu (Sử dụng Web Speech Synthesis mặc định của máy)
    // Lưu ý: Để không "giả trân", sau này bạn nên thay bằng URL file âm thanh từ Azure TTS API.
    function playSampleText() {
        const msg = new SpeechSynthesisUtterance();
        msg.text = document.getElementById('target-zh').innerText;
        msg.lang = 'zh-CN'; // Tiếng Trung Quốc đại lục
        msg.rate = 0.9; // Đọc chậm rãi truyền cảm hơn chút
        
        // Hiệu ứng nhún nhảy cho bé gấu khi nói
        const bear = document.getElementById('ai-character');
        bear.style.animation = 'bounce 0.5s infinite';
        
        window.speechSynthesis.speak(msg);
        
        msg.onend = () => {
            bear.style.animation = 'bounce 2s infinite ease-in-out';
        };
    }

    // 2. Nhận diện giọng nói và chấm điểm (Mô phỏng logic)
    let recognition;
    let isRecording = false;
    const micBtn = document.getElementById('mic-btn');

    if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        recognition = new SpeechRecognition();
        recognition.lang = 'zh-CN';
        recognition.continuous = false;
        recognition.interimResults = false;

        recognition.onstart = () => {
            isRecording = true;
            micBtn.classList.add('recording');
            document.getElementById('feedback-display').innerText = "Khu vườn đang lắng nghe bạn đọc... 🌸";
        };

        recognition.onresult = (event) => {
            const userText = event.results[0][0].transcript;
            calculateScore(userText);
        };

        recognition.onerror = (event) => {
            console.error(event);
            document.getElementById('feedback-display').innerText = "Hình như micro chưa nghe rõ, bạn thử lại nhé!";
            stopRecording();
        };

        recognition.onend = () => {
            stopRecording();
        };
    } else {
        alert("Trình duyệt của bạn không hỗ trợ nhận diện giọng nói. Hãy dùng Google Chrome nha!");
    }

    function toggleSpeechRecognition() {
        if (!recognition) return;
        if (isRecording) {
            recognition.stop();
        } else {
            recognition.start();
        }
    }

    function stopRecording() {
        isRecording = false;
        micBtn.classList.remove('recording');
    }

    // 3. Hàm chấm điểm thuật toán so sánh chuỗi cơ bản (Mô phỏng API phần trăm)
    function calculateScore(userVoiceText) {
        // Làm sạch chuỗi
        const cleanUser = userVoiceText.replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g,"");
        
        document.getElementById('feedback-display').innerHTML = `Bạn vừa đọc: "<b>${userVoiceText}</b>"`;

        // Tính toán độ khớp tương đối (Mô phỏng)
        let matchCount = 0;
        for(let char of cleanUser) {
            if(targetText.includes(char)) {
                matchCount++;
            }
        }

        let basePercent = Math.round((matchCount / targetText.length) * 100);
        if (basePercent > 100) basePercent = 100;

        // Hiển thị điểm số và lời khuyên từ Bé Gấu dễ thương
        const scoreDisplay = document.getElementById('score-display');
        const feedbackDisplay = document.getElementById('feedback-display');

        scoreDisplay.innerText = `Độ chính xác: ${basePercent}%`;

        if (basePercent >= 85) {
            scoreDisplay.style.color = "#2ecc71";
            feedbackDisplay.innerHTML += "<br>🐻 <b>Bé Gấu khen:</b> Oa! Bạn phát âm chuẩn quá, nghe cực kỳ tự nhiên luôn! Tiến lên nào!";
        } else if (basePercent >= 50) {
            scoreDisplay.style.color = "#f39c12";
            feedbackDisplay.innerHTML += "<br>🐻 <b>Bé Gấu khuyên:</b> Bạn đọc gần đúng rồi nè. Hãy chú ý bật hơi hoặc chỉnh lại thanh điệu (thanh 3 và thanh 4) một chút ở từ ở giữa nha!";
        } else {
            scoreDisplay.style.color = "#e74c3c";
            feedbackDisplay.innerHTML += "<br>🐻 <b>Bé Gấu khuyên:</b> Đừng buồn nhé! Bạn hãy ấn lại nút 🔊 để nghe kỹ giọng mẫu rồi thử lại cùng mình nha. Cố lên!";
        }
    }
</script>
